248 void QBoard::mousePressEvent(QMouseEvent *event)
249 {
250  QLabel *child = static_cast<QLabel*>(childAt(event->pos()));
251  if (!child)
252  return;
253 
254  source = child;
255  if (!child->isEnabled())
256  return;
257 
258  //if(!child->)
259 
260  if (strcmp(child->metaObject()->className(), "QLabel") != 0)
261  return;
262  QPixmap pixmap = *child->pixmap();
263 
264  QByteArray itemData;
265  QDataStream dataStream(&itemData, QIODevice::WriteOnly);
266  dataStream << pixmap << QPoint(event->pos() - child->pos());
267 
268  QMimeData *mimeData = new QMimeData;
269  mimeData->setData("application/x-dnditemdata", itemData);
270 
271  QWidget *parent = child->parentWidget();
272 
273  QDrag *drag = new QDrag(this);
274  drag->setMimeData(mimeData);
275  drag->setPixmap(pixmap);
276  drag->setHotSpot(event->pos() - child->pos() - parent->pos());
277 
278  QPixmap tempPixmap = pixmap;
279  QPainter painter;
280  painter.begin(&tempPixmap);
281  painter.fillRect(pixmap.rect(), QColor(127, 127, 127, 127));
282  painter.end();
283 
284  child->setPixmap(tempPixmap);
285 
286  if (drag->exec(Qt::CopyAction | Qt::MoveAction, Qt::CopyAction) == Qt::MoveAction) {
287  } else {
288  child->show();
289  child->setPixmap(pixmap);
290  }
291 }

502 void QBoard::on_hint_clicked()
503 {
504  std::vector<Move> moves = MoveFinder::get_available_moves(game);
505  std::string hints;
506  for (unsigned int i = 0; i < moves.size(); ++i) {
507  Move move = moves[i];
508  switch (move.get_move_type()) {
509  case STOCK_DECK_TO_WASTE_DECK:
510  hints += "Take card from stock deck to waste deck.\n";
511  break;
512  case WASTE_DECK_TO_TARGET_DECK:
513  hints += "Take card from waste deck to " + std::to_string(move.get_destination_index() + 1) + ". target deck.\n";
514  break;
515  case WASTE_DECK_TO_WORKING_STACK:
516  hints += "Take card from waste deck to " + std::to_string(move.get_destination_index() + 1) + ". working pack.\n";
517  break;
518  case TARGET_DECK_TO_WORKING_STACK:
519  hints += "Take card from " + std::to_string(move.get_source_index() + 1) + ". target deck to " + std::to_string(move.get_destination_index() + 1) + ". working stack.\n";
520  break;
521  case WORKING_STACK_TO_TARGET_DECK:
522  hints += "Take card from " + std::to_string(move.get_source_index() + 1) + ". working stack to " + std::to_string(move.get_destination_index() + 1) + ". target deck.\n";
523  break;
524  case WORKING_STACK_TO_WORKING_STACK:
525  hints += "Take card from " + std::to_string(move.get_source_index() + 1) + ". working stack to " + std::to_string(move.get_destination_index() + 1) + ". working stack since " + std::to_string(move.get_card_index() + 1) + ". card.\n";
526  break;
527  }
528  }
529 
530  QMessageBox::information(this, tr("Hints"), hints.c_str());
531 }
